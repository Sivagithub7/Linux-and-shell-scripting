create table my_table
(sno number(10),sname varchar2(100));

created

create or replace trigger my_trigger
before insert
on my_table
for each row
begin
dbms_output.put_line('my_trigger fired');
end;

Trigger MY_TRIGGER compiled

insert into my_table values(1,'raja');

1 row inserted.

my_trigger fired ---in dbms o/p

select * from my_table;

1	raja
